#include <cs50.h>
#include <stdio.h>

void print_hashes(int row);
void print_dots(int row);

int main(void){
    
    int y;
    do 
    {
     y = get_int("Height: ");
    if(y<1){
        printf("Choose a higher number \n");
    }else if(y>8){
        printf("Choose a lower number \n");
    }else{
        printf("Height: %i\n",y);
    }
    
    } while(y<1 || y>8);
    
      for (int i = 0; i < y; i++)
      {
        print_dots(i);
        print_hashes(i);
      }
      
      printf("\n");
}


void print_hashes(int row){
    row = row+1;
    for (int i = 0; i < row; i++){
    printf("#");
    }
    printf("\n");
}

void print_dots(int row){
    row=row+1;
    for (int i=0; i<(8-row); i++){
        printf(" ");
    }
}
